create view ret as(
	(select shipped_date, od.order_id, (od.unit_price*od.quantity) as total_price,
	extract(year from shipped_date) as years from orders
		inner join order_details od
			on od.order_id = orders.order_id
	where shipped_date between '1997-12-30' and '1996-01-05'
	order by shipped_date asc, order_id asc)
	union all 
	(select shipped_date, od.order_id, sum(od.unit_price*od.quantity) total_price,
	extract(year from shipped_date) as years from orders
		inner join order_details od
			on od.order_id = orders.order_id
	group by shipped_date, od.order_id, years
	order by order_id asc));
(select shipped_date, od.order_id, sum(od.unit_price*od.quantity) total_price,
	extract(year from shipped_date)::numeric as years from orders
 		inner join order_details od
 			on od.order_id = orders.order_id
 	where shipped_date between '1997-12-30' and '1998-01-05'
 	group by shipped_date, od.order_id, years
 	order by shipped_date asc, order_id asc)
union all
(select ret.shipped_date, ret.order_id, ret.total_price, ret.years::numeric
from ret
where ret.shipped_date is null and ret.total_price>4000
group by ret.shipped_date, ret.order_id, ret.years, ret.total_price)

-- 	Q8;
	alter table order_details
	add category_sales numeric;
	update order_details
	set category_sales = unit_price*quantity*(1-discount);
	select
	extract(year from orders.shipped_date) as shippedyear, 
	categories.category_name,
	round(sum(order_details.category_sales),2) as category_sales
	from  order_details
		right join orders
			on orders.order_id = order_details.order_id
		right join products
			on  order_details.product_id = products.product_id
		right join categories
			on categories.category_id = products.category_id
	where orders.shipped_date>'1997-05-31'
	group by category_name, shippedyear
	order by shippedyear;
